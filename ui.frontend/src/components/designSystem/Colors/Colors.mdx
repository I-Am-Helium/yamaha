import { Meta, ColorPalette, ColorItem, Description } from '@storybook/blocks'
import colorTokens from '../../../styles/tokens/auto_source/color.json'

<Meta title="Design System/Colors" />

# Colors

<Description of="Colors" />

export const formatTitle = (key) => {
   return key
      .split('-')
      .map((word) => word.charAt(0).toUpperCase() + word.slice(1))
      .join(' ')
}

export const getColorObject = (colorData) => {
   return Object.keys(colorData).reduce((acc, key) => {
      acc[key] = colorData[key].value
      return acc
   }, {})
}

<ColorPalette>
   {Object.keys(colorTokens.color.core.color).map((colorKey) => {
      const colorData = colorTokens.color.core.color[colorKey]
      const colors = getColorObject(colorData)

      return (
         <ColorItem
            key={colorKey}
            title={formatTitle(colorKey)}
            subtitle={colorData['500'] ? 'Base color is 500' : null}
            colors={colors}
         />
      )

})}

</ColorPalette>
